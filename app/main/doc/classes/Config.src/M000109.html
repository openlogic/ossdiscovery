  def Config.prop(key)
    if(!@@configs_loaded) then
      Config.load
    end
    if (@@configs.has_key?(key.to_sym)) then
      return @@configs[key.to_sym]
    else
      raise "A property does not exist for the key arg '#{key}'. Valid keys: #{@@configs.keys.inspect}"
    end
  end
