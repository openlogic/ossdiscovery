  def initialize(base_url)
    if (base_url == nil) then
      @base_url = "http://localhost:3000/"
    else
      if (base_url[base_url.size - 1 ..base_url.size - 1] == "/")
        @base_url = base_url
      else
        @base_url = base_url << "/"
      end
    end
    
    @proxy_host = nil
    @proxy_port = nil
    @proxy_username = nil
    @proxy_password = nil
    
  end
