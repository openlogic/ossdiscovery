  def BooleanExpression.get_operands(the_expression="")
    if (the_expression == nil) then 
      the_expression = ""
    end
    # boil the expression down to just the operands that are in it
    operands = "" << the_expression
    operands.gsub!("(","")
    operands.gsub!(")","")
    operands.gsub!("AND", "")
    operands.gsub!("OR", "")
    operands.gsub!("NOT", "")
    
    return operands.split(" ")
  end
