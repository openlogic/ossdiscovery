  def initialize(name, from="wild", operating_systems=Set["all"]) 
    @name = name
    @from = from
    if (operating_systems != nil) then
      @operating_systems = operating_systems.to_set
    else 
      @operating_systems = nil
    end
    
    @rulesets = Set.new
    @eval_rules = Set.new
    @desc = "not yet defined"
  end
