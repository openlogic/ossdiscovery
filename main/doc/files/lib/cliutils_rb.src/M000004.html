def report_audit_records( records )
  io = nil
  if (@results == STDOUT) then
    io = STDOUT
  else 
    io = File.new(@results, "w")
  end
  
  r_strings = Array.new
  records.each {|r| r_strings << r.to_s}
  r_strings.sort!
  
  r_strings.each {|r| printf(io, r.to_s + "\n")}
  
  if (io != STDOUT) then io.close end
end
